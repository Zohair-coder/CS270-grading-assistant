(define (running-list E L)
 (if (list? E)
   (if (empty? E)
     L

     (if (list? (first E))
       (running-list (rest E) (running-list (first E) L))
       (if (is-reserved-word? (first E)) (running-list (rest E) L) (running-list (rest E) (cons (first E) L)))
     )
   )
   (cons E L)
  )
)
(define (get-variables expression) (reverse (remove-duplicates (running-list expression '()))))
