(define (get-variables expression)
  (cond
    [(null? expression) '()]
    [(not (list? expression)) (cons expression '())]
    [(is-variable? (first expression)) (cons (first expression) (get-variables (rest expression)))]
    [(list? (first expression)) (remove-duplicates (get-variables (append (first expression) (rest expression))))]
    [else (get-variables (rest expression))]
    )
)
