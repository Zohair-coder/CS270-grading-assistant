(define (get-var-helper expression var-list)
  (cond
    [(null? expression) var-list]
    [(is-constant? expression) null]
    [(is-variable? expression) (cons expression var-list)]
    [(is-not? expression) (get-var-helper (second expression) var-list)]
    [else (append (get-var-helper (second expression) var-list) (get-var-helper (third expression) var-list))]
    )
  )

(define (get-variables expression)
  (remove-duplicates (get-var-helper expression '()))
