(define (gvhelp expression L)
 (if (list? expression)
   (if (empty? expression)
     L
     (if (list? (first expression))
       (gvhelp (rest expression) (gvhelp (first expression) L))
       (if (is-reserved-word? (first expression))
           (gvhelp (rest expression) L)
           (gvhelp (rest expression) (append L (list (first expression))))
           )
     )
   )
   (append L (list expression))
  )
)

(define (get-variables expression)
  (remove-duplicates (gvhelp expression '()))
  )
