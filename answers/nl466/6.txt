(define (get-variables expression)
  (cond
    [(is-variable? expression) (list expression)]
    [(is-constant? expression) '()]
    [(is-reserved-word? (first expression)) (remove-duplicates (if (is-not? expression) (get-variables (first (rest expression))) (append (get-variables (first (rest expression))) (get-variables (first (rest (rest expression)))))))]
    [else '()]
    )
)
